zDropdown

基本使用

new zDropdown({
	el: "myDropdown"
})

回调事件

new zDropdown({
	el: document.getElementById("myDropdownEvent"),
	change: function(value, text, index) {
		alert("value: " + value + ", text: " + text + ", index: " + index);
	}
})

销毁

var myDropdown = new zDropdown({
	el: document.getElementById("myDropdownDestroyEvent"),
	change: function(value, text, index) {
		alert("value: " + value + ", text: " + text + ", index: " + index);
	}
});
document.getElementById("destroyMyDropdownEvent").onclick = function() {
	myDropdown.destroy();
}

手动触发选择项

眼保健操 早操 午休 早读 
var manualDropdown = new zDropdown({
	el: document.getElementById("manualDropdown"),
	change: function(value, text, index) {
		console.log("value: " + value + ", text: " + text + ", index: " + index);
	}
});
manualDropdown.changeItem(index);

自定义皮肤

new zDropdown({
	el: document.getElementById("mySkinDropdown"),
	skin: "my-skin"
})
